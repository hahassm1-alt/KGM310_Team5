# 2. 전처리 (Unnamed 열 제거 및 공백 정리)
df = df.loc[:, ~df.columns.str.contains('^Unnamed')]
df.columns = df.columns.str.strip()

# 3. 한글 컬럼명을 영문으로 변경
translation_dict = {
    '구분': 'Region',
    '이송인원': 'Total_Transported',
    '10세 미만': 'Under 10s',
    '10대': '10s',
    '20대': '20s',
    '30대': '30s',
    '40대': '40s',
    '50대': '50s',
    '60대': '60s',
    '70대': '70s',
    '80대': '80s',
    '90대': '90s',
    '100세 이상': 'Over 100s'
}
df_eng = df.rename(columns=translation_dict)

# 4. 'Total(합계)' 행 데이터 추출
df_total = df_eng[df_eng['Region'].str.contains('합계', na=False)]

# X축(연령대)과 Y축(인원수) 설정
age_categories = ['Under 10s', '10s', '20s', '30s', '40s', '50s', '60s', '70s', '80s', '90s', 'Over 100s']
values = df_total[age_categories].iloc[0].values

# --- 시각화 (Line Chart) ---

sns.set_theme(style="darkgrid") # 선 그래프는 격자가 있는 스타일이 보기 좋습니다.
plt.figure(figsize=(12, 6))

# 선 그래프 그리기 (marker를 추가하여 각 포인트를 강조)
plt.plot(age_categories, values, marker='o', linestyle='-', color='b', linewidth=2, markersize=8)

# 그래프 제목 및 라벨 설정 (English)
plt.title('Trend of Emergency Transports by Age Group', fontsize=16, pad=20)
plt.xlabel('Age Group', fontsize=12)
plt.ylabel('Number of People', fontsize=12)

# 각 포인트 위에 숫자 표시 (선택 사항)
for i, txt in enumerate(values):
    plt.annotate(format(txt, ',.0f'), (age_categories[i], values[i]), 
                 textcoords="offset points", xytext=(0,10), ha='center', fontsize=9)

plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
